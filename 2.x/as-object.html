# As object

# Method provided

Lists all method provided by the trait.

# make

Resolves the action from the container.

MyAction::make();

// Equivalent to:
app(MyAction::class);

# run

Resolves and executes the action.

MyAction::run($someArguments);

// Equivalent to:
MyAction::make()->handle($someArguments);
